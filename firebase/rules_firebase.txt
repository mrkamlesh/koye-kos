rules_version = '2';
service cloud.firestore {
  match /databases/{databases}/documents {

    match /camps/{camp} {
      allow read: if true;
      allow write: if request.auth.token.firebase.sign_in_provider in ['google.com'];
    }

    match /users/{id}/{document=**} {
      allow read, write: if request.auth.uid == id;
    }
  }
}
